    <script>
        const initialPositions = {
            'black': {
                '車': [[0, 0], [8, 0]],
                '馬': [[1, 0], [7, 0]],
                '相': [[2, 0], [6, 0]],
                '士': [[3, 0], [5, 0]],
                '將': [[4, 0]],
                '炮': [[1, 2], [7, 2]],
                '卒': [[0, 3], [2, 3], [4, 3], [6, 3], [8, 3]]
            },
            'red': {
                '車': [[0, 9], [8, 9]],
                '馬': [[1, 9], [7, 9]],
                '象': [[2, 9], [6, 9]],
                '仕': [[3, 9], [5, 9]],
                '帥': [[4, 9]],
                '炮': [[1, 7], [7, 7]],
                '兵': [[0, 6], [2, 6], [4, 6], [6, 6], [8, 6]]
            }
        };

        const board = document.getElementById('board');
        const startMessage = document.getElementById('start-message');
        const startButton = document.getElementById('start-button');

        const state = {
            selectedPiece: null,
            turn: 'red',
            gameStatus: 'waitingToStart',
        };

        function createCell(x, y) {
            const cell = document.createElement('div');
            cell.className = 'cell';
            if (y === 4 || y === 5) {
                cell.classList.add('black');
            }
            cell.x = x;
            cell.y = y;
            cell.cellIndex = y * 9 + x;
            return cell;
        }

        function createPiece(name, color) {
            const piece = document.createElement('div');
            piece.className = 'piece';
            piece.textContent = name;
            piece.style.color = color === 'black' ? 'black' : 'red';
            piece.dataset.color = color;
            piece.dataset.name = name;
            return piece;
        }

        function initializeBoard() {
            for (let y = 0; y < 10; y++) {
                for (let x = 0; x < 9; x++) {
                    const cell = createCell(x, y);
                    board.appendChild(cell);
                }
            }
            for (const [color, pieces] of Object.entries(initialPositions)) {
                for (const [name, positions] of Object.entries(pieces)) {
                    positions.forEach(([x, y]) => {
                        const piece = createPiece(name, color);
                        const cell = board.children[y * 9 + x];
                        cell.appendChild(piece);
                        piece.style.top = `${cell.offsetTop}px`;
                        piece.style.left = `${cell.offsetLeft}px`;
                    });
                }
            }
        }

        function isValidMove(piece, fromX, fromY, toX, toY) {
            const name = piece.dataset.name;
            const color = piece.dataset.color;
            const dx = Math.abs(fromX - toX);
            const dy = Math.abs(fromY - toY);

            switch (name) {
                case '車':
                    if (fromX !== toX && fromY !== toY) return false;
                    if (fromX === toX) {
                        const [min, max] = fromY < toY ? [fromY, toY] : [toY, fromY];
                        for (let i = min + 1; i < max; i++) {
                            if (board.children[i * 9 + fromX].hasChildNodes()) return false;
                        }
                    } else {
                        const [min, max] = fromX < toX ? [fromX, toX] : [toX, fromX];
                        for (let i = min + 1; i < max; i++) {
                            if (board.children[fromY * 9 + i].hasChildNodes()) return false;
                        }
                    }
                    break;
                case '馬':
                    if (!((dx === 2 && dy === 1) || (dx === 1 && dy === 2))) return false;
                    if (dx === 2) {
                        const midX = (fromX + toX) / 2;
                        if (board.children[fromY * 9 + midX].hasChildNodes()) return false;
                    } else {
                        const midY = (fromY + toY) / 2;
                        if (board.children[midY * 9 + fromX].hasChildNodes()) return false;
                    }
                    break;
                case '相':
                case '象':
                    if (dx !== 2 || dy !== 2) return false;
                    if (color === 'black' && toY > 4) return false;
                    if (color === 'red' && toY < 5) return false;
                    const midX = (fromX + toX) / 2;
                    const midY = (fromY + toY) / 2;
                    if (board.children[midY * 9 + midX].hasChildNodes()) return false;
                    break;
                case '士':
                case '仕':
                    if (dx !== 1 || dy !== 1) return false;
                    if (color === 'black' && (toX < 3 || toX > 5 || toY > 2)) return false;
                    if (color === 'red' && (toX < 3 || toX > 5 || toY < 7)) return false;
                    break;
                case '將':
                case '帥':
                    if (dx + dy !== 1) return false;
                    if (color === 'black' && (toX < 3 || toX > 5 || toY > 2)) return false;
                    if (color === 'red' && (toX < 3 || toX > 5 || toY < 7)) return false;
                    break;
                case '炮':
                    if (fromX !== toX && fromY !== toY) return false;
                    if (fromX === toX) {
                        const [min, max] = fromY < toY ? [fromY, toY] : [toY, fromY];
                        let piecesInBetween = 0;
                        for (let i = min + 1; i < max; i++) {
                            if (board.children[i * 9 + fromX].hasChildNodes()) piecesInBetween++;
                        }
                        if (piecesInBetween === 1) return board.children[toY * 9 + toX].hasChildNodes();
                        return piecesInBetween === 0;
                    } else {
                        const [min, max] = fromX < toX ? [fromX, toX] : [toX, fromX];
                        let piecesInBetween = 0;
                        for (let i = min + 1; i < max; i++) {
                            if (board.children[fromY * 9 + i].hasChildNodes()) piecesInBetween++;
                        }
                        if (piecesInBetween === 1) return board.children[toY * 9 + toX].hasChildNodes();
                        return piecesInBetween === 0;
                    }
                case '卒':
                case '兵':
                    if (color === 'black') {
                        if (fromY < 5) {
                            if (dx + dy !== 1) return false;
                            if (dy === 1 && dx === 0) return fromY < toY;
                            return dx === 1 && dy === 0;
                        } else {
                            return dx === 0 && toY - fromY === 1;
                        }
                    } else {
                        if (fromY > 4) {
                            if (dx + dy !== 1) return false;
                            if (dy === 1 && dx === 0) return toY < fromY;
                            return dx === 1 && dy === 0;
                        } else {
                            return dx === 0 && fromY - toY === 1;
                        }
                    }
            }
            return true;
        }

        function movePiece(piece, toX, toY) {
            const fromX = piece.parentElement.x;
            const fromY = piece.parentElement.y;
            const cell = board.children[toY * 9 + toX];
            cell.appendChild(piece);
            piece.style.top = `${cell.offsetTop}px`;
            piece.style.left = `${cell.offsetLeft}px`;
        }

        function getValidMoves(piece) {
            const fromX = piece.parentElement.x;
            const fromY = piece.parentElement.y;
            const validMoves = [];

            for (let toY = 0; toY < 10; toY++) {
                for (let toX = 0; toX < 9; toX++) {
                    if ((toX !== fromX || toY !== fromY) && isValidMove(piece, fromX, fromY, toX, toY)) {
                        validMoves.push([toX, toY]);
                    }
                }
            }

            return validMoves;
        }

        function getRandomMove(piece) {
            const validMoves = getValidMoves(piece);
            if (validMoves.length === 0) return null;
            const randomIndex = Math.floor(Math.random() * validMoves.length);
            return validMoves[randomIndex];
        }

        async function aiMove(color) {
            const pieces = Array.from(board.getElementsByClassName('piece'))
                .filter(piece => piece.dataset.color === color);
            let moved = false;
            while (!moved) {
                const piece = pieces[Math.floor(Math.random() * pieces.length)];
                const move = getRandomMove(piece);
                if (move) {
                    movePiece(piece, move[0], move[1]);
                    moved = true;
                    await new Promise(resolve => setTimeout(resolve, 1000)); // Delay for visualizing the move
                }
            }
        }

        async function startGame() {
            startMessage.style.visibility = 'hidden';
            startButton.style.visibility = 'hidden';
            board.style.visibility = 'visible';
            state.gameStatus = 'inProgress';

            while (state.gameStatus === 'inProgress') {
                await aiMove(state.turn);
                state.turn = state.turn === 'red' ? 'black' : 'red';
                // Add condition to check for end of the game and set state.gameStatus to 'ended'
                // For simplicity, this demo does not implement checkmate detection.
            }
        }

        startButton.addEventListener('click', startGame);
        initializeBoard();
    </script>
